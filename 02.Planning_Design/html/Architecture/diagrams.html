📂 프로젝트 폴더 및 패키지 구조

Frontend: FSD (Feature-Sliced Design)

src/
 ├── app/             # 전역 라우터, 글로벌 스타일, 프로바이더
 ├── assets/          # 로고 이미지, 폰트, 공통 CSS
 ├── features/        # 도메인 중심 핵심 기능 모듈
 │    ├── auth/       # 프로필 설정, 21종 알레르기 온보딩
 │    ├── scanner/    # AI 렌즈 뷰, 상태 관리(Zustand)
 │    ├── order/      # 점주용 웹 오더 뷰어 (QR)
 │    ├── tour/       # 식당 지도, 리뷰 필터
 │    ├── cs/         # 게시판, FAQ, 1:1 문의, 개인정보
 │    └── admin/      # 관리자용 통계 모니터링 대시보드
 ├── shared/          # 공통 컴포넌트(Button), API 클라이언트
 └── App.jsx

Backend: Layered Multi-Module

okeymeal-backend/
 ├── okeymeal-core/          # 의존성 없음, 핵심 도메인 로직
 │    ├── domain/            # user, tour, ai, order, cs 등
 │    └── global/            # 예외 처리, 보안(JWT)
 │
 ├── okeymeal-external/      # 외부 API 전용 클라이언트
 │    ├── ai/                # Naver OCR, HyperCLOVA X
 │    └── tour/              # 한국관광공사 TourAPI 연동
 │
 ├── okeymeal-api/           # 관광객(B2C) 및 점주(B2B) API
 │    ├── controller/        # REST Endpoint
 │    └── dto/               # Request/Response 객체
 │
 └── okeymeal-admin/         # 관리자(B2A) API 및 스케줄러
      ├── stats/             # 일간/주간 통계 집계 배치
      └── controller/        # 관리자 대시보드 데이터 연동
